简单灯泡开关

<meta charset="UTF-8">
电灯开关 
<img id="light" src="img/off.gif">



<script>

  /* 

     1.获取图片对象

     2.绑定事件

     3.每次点击切换图片

        灯开则关；灯关则开

 */

  var light=document.getElementById("light");

  var flag=false;//设置flag判断灯目前的状态是开还是关

  light.onclick=function(){

      if(flag==false){

        light.src="img/on.gif";

       flag=true;

      }

      else{

        light.src="img/off.gif";

       flag=false;

      }



  }



</script>
